\section{Zahlentheorie}

\subsection{Division mit Rest}
$ A = q * n + r$ wobei $ 0 \leq r < |n|$

\subsection{Kongruenz modulo n}
\begin{tabular}{lll}
    $ a \equiv b (mod n)$ & $\iff$ & $n|(a - b)$ \\
    & $\iff$ & $\exists q : a-b = q * n$ \\
    & $\iff$ & $\exists q : a = b + q * n$ \\
\end{tabular}
